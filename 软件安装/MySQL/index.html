MySQL安装

源码安装

安装依赖

yum -y install ncurses-devel install gcc-* bzip2-* bison

安装cmake

tar -xvf cmake-3.6.0-rc1.tar.gz
cd cmake-3.6.0-rc1
./configure
make && make install

安装Boost库

tar -xvf boost_1_59_0.tar.bz2
mv ./boost_1_59_0 /usr/local/boost/

安装MySQL

tar -xvf mysql-5.7.24.tar.gz
cd mysql-5.7.24

cmake编译

cmake . -DCMAKE_INSTALL_PREFIX=/usr/local/mysql  \
DMYSQL_DATADIR=/usr/local/mysql/data/ \
DMYSQL_UNIX_ADDR=/usr/local/mysql/mysql.sock \
DWITH_INNOBASE_STORAGE_ENGINE=1 \
DWITH_MYISAM_STORAGE_ENGINE=1 \
DENABLED_LOCAL_INFILE=1 \
DEXTRA_CHARSETS=all -DDEFAULT_CHARSET=utf8 -DDEFAULT_COLLATION=utf8_general_ci \
DWITH_DEBUG=0 \
DWITH_EMBEDDED_SERVER=1 \
DDOWNLOAD_BOOST=1 -DENABLE_DOWNLOADS=1 -DWITH_BOOST=/usr/local/boost

编译

make

安装

make install

配置

cp /usr/local/mysql/support-files/mysql.server /etc/init.d/mysql
chmod 755 /etc/init.d/mysql
useradd -s /sbin/nologin -r mysql
chown mysql.mysql  /usr/local/mysql/ -R
ln -sf /usr/local/mysql/bin/* /usr/bin/
ln -sf /usr/local/mysql/lib/* /usr/lib/
ln -sf /usr/local/mysql/libexec/* /usr/libexec/
ln -sf /usr/local/mysql/share/man/man1/* /usr/share/man/man1
ln -sf /usr/local/mysql/share/man/man8/* /usr/share/man/man8

vim /etc/my.cnf

[mysqld]
basedir=/usr/local/mysql
datadir=/usr/local/mysql/data
socket=/usr/local/mysql/mysql.sock
symbolic-links=0

[mysqld_safe]
log-error=/var/log/mysql.log
pid-file=/var/run/mysql.pid

启动

/etc/init.d/mysql start

初始化

/usr/local/mysql/bin/mysqld --initialize --user=mysql --basedir=/usr/local/mysql/ --datadir=/usr/local/mysql/data/

mysql_secure_installation

mysql -u root -p

使用

cat /var/log/mysqld.log|grep "temporary password"

连接

mysql -u root -p

修改密码

#查看当前密码复杂度策略
show variables like 'validate_password%';

#修改密码复杂度策略
set global validate_password.policy=0;
set global validate_password.mixed_case_count=0;
set global validate_password.number_count=0;
set global validate_password.special_char_count=0;
set global validate_password.length=1;

#设置密码（使用mysql_native_password认证插件）
ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password by 'mysql';

#刷新策略
flush privileges;

允许root用户远程访问

use mysql;
update user set host = '%' where user = 'root';
grant all privileges on *.* to 'root'@'%';

二进制安装

下载地址：https://downloads.mysql.com/archives/community/

参考：https://dev.mysql.com/doc/refman/8.4/en/binary-installation.html

安装依赖

yum install -y libaio

apt update
apt-get install libaio-dev libaio1

安装

tar -xf mysql-8.0.31-linux-glibc2.17-aarch64.tar.gz -C /usr/local/
cd /usr/local
mv mysql-8.0.31-linux-glibc2.17-aarch64 mysql
mkdir -p mysql-files conf logs tmp
chown -R mysql:mysql /usr/local/mysql/logs /usr/local/mysql/tmp

groupadd mysql
useradd -r -g mysql -s /bin/false mysql
chown mysql:mysql mysql-files
chmod 750 mysql-files
#初始化
mysqld --initialize --user=mysql

配置环境变量

echo "export PATH=/usr/local/mysql/bin:$PATH" >> /etc/profile
echo "export MYSQL_UNIX_PORT=/usr/local/mysql/tmp/mysqld.sock" >> /etc/profile
source /etc/profile

配置文件

/usr/local/mysql/conf/my.cnf

[mysqld]
basedir= /usr/local/mysql
datadir= /usr/local/mysql/data
socket= /usr/local/mysql/tmp/mysqld.sock
mysqlx_socket = /usr/local/mysql/tmp/mysqlx.sock
symbolic-links= 0
bind-address = 0.0.0.0
port = 3306
user = mysql
log-error= /usr/local/mysql/logs/mysqld.log
pid-file= /usr/local/mysql/tmp/mysqld.pid
tmpdir = /usr/local/mysql/tmp

[client]
socket = /usr/local/mysql/tmp/mysqld.sock

启动

mysqld --defaults-file=/usr/local/mysql/conf/my.cnf

关闭

mysqladmin -u root -p shutdown

注册到系统服务

vi /etc/systemd/system/mysqld.service

[Unit]
Description=MySQL Server
After=network.target

[Service]
User=mysql
Group=mysql
ExecStart=/usr/local/mysql/bin/mysqld --defaults-file=/usr/local/mysql/conf/my.cnf
ExecStop=/usr/local/mysql/bin/mysqladmin shutdown
Restart=on-failure
RestartSec=5s

[Install]
WantedBy=multi-user.target

操作

systemctl daemon-reload
systemctl start mysqld
systemctl stop mysqld
