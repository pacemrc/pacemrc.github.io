Weblogic安装

安装Java

安装Weblogic

创建用户及目录

sudo mkdir -p /usr/local/weblogic/middleware
sudo mkdir -p /usr/local/weblogic/oraInventory
useradd weblogic
passwd weblogic
sudo chown -R weblogic:weblogic /usr/local/weblogic

创建响应文件

cat << EOF >> /tmp/weblogic_install.rsp
[ENGINE]
#DO NOT CHANGE THIS.
Response File Version=1.0.0.0.0
[GENERIC]
#The oracle home location. This can be an existing Oracle Home or a new Oracle Home
ORACLE_HOME=/usr/local/weblogic/middleware
#Set this variable value to the Installation Type selected. e.g. WebLogic Server, Coherence, Complete with Examples.
INSTALL_TYPE=WebLogic Server
#Provide the My Oracle Support Username. If you wish to ignore Oracle Configuration Manager configuration provide empty string for user name.
MYORACLESUPPORT_USERNAME=
#Provide the My Oracle Support Password
MYORACLESUPPORT_PASSWORD=<SECURE VALUE>
#Set this to true if you wish to decline the security updates. Setting this to true and providing empty string for My Oracle Support username will ignore the Oracle Configuration Manager configuration
DECLINE_SECURITY_UPDATES=true
#Set this to true if My Oracle Support Password is specified
SECURITY_UPDATES_VIA_MYORACLESUPPORT=false
#Provide the Proxy Host
PROXY_HOST=
#Provide the Proxy Port
PROXY_PORT=
#Provide the Proxy Username
PROXY_USER=
#Provide the Proxy Password
PROXY_PWD=<SECURE VALUE>
#Type String (URL format) Indicates the OCM Repeater URL which should be of the format [scheme[Http/Https]]://[repeater host]:[repeater port]
COLLECTOR_SUPPORTHUB_URL=
EOF

创建oraInst.loc文件

Oracle 产品清单（Oracle Inventory） 的路径配置，主要用于记录 Oracle 软件（如 WebLogic、数据库等）的安装信息

cat << EOF >> /tmp/weblogic_oraInst.loc
inventory_loc=/usr/local/weblogic/oraInventory
inst_group=oinstall
EOF

创建域响应文件

cat << EOF >> /tmp/weblogic_domain.py
readTemplate('/usr/local/weblogic/middleware/wlserver/common/templates/wls/wls.jar')
cd('Servers/AdminServer')
set('ListenPort', 7001)
set('Name', 'AdminServer')
cd('/Security/base_domain/User/weblogic')
cmo.setPassword('Jowto@2307')
cd('/')
setOption('OverwriteDomain', 'true')
writeDomain('/usr/local/weblogic/middleware/user_projects/domains/base_domain')
closeTemplate()
EOF

执行静默安装

使安装加速

sed -i 's/securerandom\.source=file:\/dev\/random/securerandom\.source=file:\/dev\/\.\/urandom/g' "$JAVA_HOME/jre/lib/security/java.security"

root用户

mv fmw_12.2.1.3.0_wls.jar /home/weblogic

使用weblogic用户登录

java -jar fmw_12.2.1.3.0_wls.jar -silent -responseFile /tmp/weblogic_install.rsp -invPtrLoc /tmp/weblogic_oraInst.loc

执行创建域命令

/usr/local/weblogic/middleware/wlserver/common/bin/wlst.sh /tmp/weblogic_domain.py

启动

cd /usr/local/weblogic/middleware/user_projects/domains/base_domain/bin
./startWebLogic.sh
