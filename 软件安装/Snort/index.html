Snort安装

安装

ubuntu 22.04

snort-2.9.20

apt update
apt install vim wget iproute2 build-essential libpcap-dev libpcre3-dev libnet1-dev zlib1g-dev luajit hwloc libdnet-dev libdumbnet-dev bison flex liblzma-dev openssl libssl-dev pkg-config libhwloc-dev cmake cpputest libsqlite3-dev uuid-dev libcmocka-dev libnetfilter-queue-dev libmnl-dev autotools-dev libluajit-5.1-dev libunwind-dev libtirpc-dev
wget https://www.snort.org/downloads/snort/daq-2.0.7.tar.gz
tar -xf daq-2.0.7.tar.gz
cd daq-2.0.7
./configure
make -j 4
make install
cd /opt/
wget https://www.snort.org/downloads/snort/snort-2.9.20.tar.gz
tar -xf snort-2.9.20.tar.gz
cd snort-2.9.20
cp -r  /usr/include/tirpc/rpc/  /usr/include/
cp -r /usr/include/tirpc/netconfig.h  /usr/include/
./configure --prefix=/opt/snort2_9 --enable-sourcefire    #安装在/opt/snort2_9
make -j 4
make install
ldconfig

创建配置目录及文件

mkdir /opt/snort2_9/etc
mkdir /opt/snort2_9/rules
mkdir /opt/snort2_9/preproc_rules
mkdir /opt/snort2_9/lib/snort_dynamicrules
mkdir /opt/snort2_9/so_rules
mkdir /opt/snort2_9/log
mkdir /opt/snort2_9/log/archived_logs
touch /opt/snort2_9/rules/black_list.rules
touch /opt/snort2_9/rules/white_list.rules
touch /opt/snort2_9/rules/local.rules
touch /opt/snort2_9/sid-msg.map

cp /opt/snort-2.9.20/etc/*.conf* /opt/snort2_9/etc/
cp /opt/snort-2.9.20/etc/*.map /opt/snort2_9/etc/
cp /opt/snort-2.9.20/etc/*.dtd /opt/snort2_9/etc/
cp /opt/snort-2.9.20/src/dynamic-preprocessors/build/opt/snort2_9/lib/snort_dynamicpreprocessor/* /opt/snort2_9/lib/snort_dynamicpreprocessor/

修改配置文件

#修改库文件路径
sed -i 's/\.\./\/opt\/snort2\_9/' /opt/snort2_9/etc/snort.conf
sed -i 's/\/usr\/local/\/opt\/snort2_9/' /opt/snort2_9/etc/snort.conf
#设置取消导入所有规则文件
sed -i 's/include \$RULE\_PATH/#include \$RULE\_PATH/' /opt/snort2_9/etc/snort.conf
#设置仅启用本地规则文件
sed -i 's/#include \$RULE\_PATH\/local.rules/include \$RULE\_PATH\/local.rules/' /opt/snort2_9/etc/snort.conf

验证配置

snort -T -c /opt/snort2_9/etc/snort.conf -i eth0

匹配数据包

cd /opt/snort2_9
snort -A console -q -c etc/snort.conf -r pcaps/CVE-2023-28432.pcapng -l log
