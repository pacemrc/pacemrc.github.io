GitLab搭建

安装gitlab-yum源

curl -s https://packages.gitlab.com/install/repositories/gitlab/gitlab-ce/script.rpm.sh | sudo bash

安装依赖

yum -y install policycoreutils openssh-server postfix
yum install policycoreutils-python
systemctl enable sshd && sudo systemctl start sshd
systemctl enable postfix && systemctl start postfix

下载RPM安装包

wget --content-disposition https://packages.gitlab.com/gitlab/gitlab-ce/packages/el/7/gitlab-ce-16.2.7-ce.0.el7.x86_64.rpm/download.rpm --no-check-certificate

安装

yum install -y gitlab-ce-16.2.7-ce.0.el7.x86_64.rpm

修改配置

vim /etc/gitlab/gitlab.rb

external_url 'http://10.58.120.200'

重新启动

gitlab-ctl reconfigure
gitlab-ctl restart
